<script>
function check(answer) {
  let message = "";

  if (answer === "never") {
    message = "⚠️ You have never screened. Please visit a health facility immediately.";
  } 
  else if (answer === "1year") {
    message = "✅ Good! Continue regular screening as advised.";
  } 
  else if (answer === "3years") {
    message = "⚠️ You are due for screening. Visit a clinic soon.";
  } 
  else {
    message = "🚨 You need screening urgently. Please go to a hospital.";
  }

  document.getElementById("result").innerText = message;
}
</script>
